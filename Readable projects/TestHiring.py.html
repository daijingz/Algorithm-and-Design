import unittest
import Hiring


class MyTestCase(unittest.TestCase):
    def test_eq(self):
        """! Testing of __eq__ method """
        hiring1 = Hiring.Hiring(100, 20, lambda x: x + 50, 300, 50)
        hiring2 = Hiring.Hiring(100, 20, lambda x: x + 30, 300, 50)
        self.assertFalse(hiring1 == hiring2)
